达梦建表空间和用户

```
create tablespace ricisso datafile 'ricisso.dbf' size 100 autoextend on maxsize 500;

create user uricisso identified by uricisso123 default tablespace ricisso;

grant dba to uricisso;

create tablespace ricicms datafile 'ricicms.dbf' size 100 autoextend on maxsize 500;

create user uricicms identified by uricicms123 default tablespace ricicms;

grant dba to uricicms;
```
